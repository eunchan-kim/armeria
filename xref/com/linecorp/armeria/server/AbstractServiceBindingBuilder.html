20  import static com.linecorp.armeria.server.ServerConfig.validateRequestTimeoutMillis;
21  import static java.util.Objects.requireNonNull;
22  
23  import java.nio.charset.Charset;
24  import java.time.Duration;
25  import java.util.List;
26  import java.util.function.Function;
27  
28  import javax.annotation.Nullable;
29  
30  import com.linecorp.armeria.common.HttpRequest;
31  import com.linecorp.armeria.common.HttpResponse;
32  import com.linecorp.armeria.common.logging.ContentPreviewer;
33  import com.linecorp.armeria.common.logging.ContentPreviewerFactory;

37  /**
38   * A builder class for binding a {@link Service} fluently.
39   *
40   * @see ServiceBindingBuilder
41   * @see VirtualHostServiceBindingBuilder
42   */
43  abstract class AbstractServiceBindingBuilder extends AbstractBindingBuilder {
44  
45      @Nullable
46      private Long requestTimeoutMillis;
47      @Nullable
48      private Long maxRequestLength;
49      @Nullable
50      private Boolean verboseResponses;
51      @Nullable
52      private ContentPreviewerFactory requestContentPreviewerFactory;
53      @Nullable
54      private ContentPreviewerFactory responseContentPreviewerFactory;
55      @Nullable
56      private Function<Service<HttpRequest, HttpResponse>, Service<HttpRequest, HttpResponse>> decorator;
57      @Nullable
58      private AccessLogWriter accessLogWriter;
59      private boolean shutdownAccessLogWriterOnStop;
60  
61      /**
62       * Sets the timeout of an HTTP request. If not set, {@link VirtualHost#requestTimeoutMillis()}
63       * is used.
64       *
65       * @param requestTimeout the timeout. {@code 0} disables the timeout.
66       */
67      public AbstractServiceBindingBuilder requestTimeout(Duration requestTimeout) {
68          return requestTimeoutMillis(requireNonNull(requestTimeout, "requestTimeout").toMillis());
69      }
70  
71      /**
72       * Sets the timeout of an HTTP request in milliseconds. If not set,
73       * {@link VirtualHost#requestTimeoutMillis()} is used.
74       *
75       * @param requestTimeoutMillis the timeout in milliseconds. {@code 0} disables the timeout.
76       */
77      public AbstractServiceBindingBuilder requestTimeoutMillis(long requestTimeoutMillis) {
78          this.requestTimeoutMillis = validateRequestTimeoutMillis(requestTimeoutMillis);
79          return this;
80      }
81  
82      /**
83       * Sets the maximum allowed length of an HTTP request. If not set,
84       * {@link VirtualHost#maxRequestLength()} is used.
85       *
86       * @param maxRequestLength the maximum allowed length. {@code 0} disables the length limit.
87       */
88      public AbstractServiceBindingBuilder maxRequestLength(long maxRequestLength) {
89          this.maxRequestLength = validateMaxRequestLength(maxRequestLength);
90          return this;
91      }
92  
93      /**
94       * Sets whether the verbose response mode is enabled. When enabled, the service response will contain
95       * the exception type and its full stack trace, which may be useful for debugging while potentially
96       * insecure. When disabled, the service response will not expose such server-side details to the client.
97       * If not set, {@link VirtualHost#verboseResponses()} is used.
98       */
99      public AbstractServiceBindingBuilder verboseResponses(boolean verboseResponses) {
100         this.verboseResponses = verboseResponses;
101         return this;
102     }
103 
104     /**
105      * Sets the {@link ContentPreviewerFactory} for an HTTP request of a {@link Service}.
106      * If not set, {@link VirtualHost#requestContentPreviewerFactory()}
107      * is used.
108      */
109     public AbstractServiceBindingBuilder requestContentPreviewerFactory(ContentPreviewerFactory factory) {
110         requestContentPreviewerFactory = requireNonNull(factory, "factory");
111         return this;
112     }
113 
114     /**
115      * Sets the {@link ContentPreviewerFactory} for an HTTP response of a {@link Service}.
116      * If not set, {@link VirtualHost#responseContentPreviewerFactory()}
117      * is used.
118      */
119     public AbstractServiceBindingBuilder responseContentPreviewerFactory(ContentPreviewerFactory factory) {
120         responseContentPreviewerFactory = requireNonNull(factory, "factory");
121         return this;
122     }
123 
124     /**
125      * Sets the {@link ContentPreviewerFactory} for creating a {@link ContentPreviewer} which produces the
126      * preview with the maximum {@code length} limit for an HTTP request/response of the {@link Service}.
127      * The previewer is enabled only if the content type of an HTTP request/response meets
128      * any of the following conditions:
129      * <ul>
130      *     <li>when it matches {@code text/*} or {@code application/x-www-form-urlencoded}</li>
131      *     <li>when its charset has been specified</li>
132      *     <li>when its subtype is {@code "xml"} or {@code "json"}</li>
133      *     <li>when its subtype ends with {@code "+xml"} or {@code "+json"}</li>
134      * </ul>
135      * @param length the maximum length of the preview.
136      */
137     public AbstractServiceBindingBuilder contentPreview(int length) {
138         return contentPreview(length, ArmeriaHttpUtil.HTTP_DEFAULT_CONTENT_CHARSET);
139     }
140 
141     /**
142      * Sets the {@link ContentPreviewerFactory} for creating a {@link ContentPreviewer} which produces the
143      * preview with the maximum {@code length} limit for an HTTP request/response of a {@link Service}.
144      * The previewer is enabled only if the content type of an HTTP request/response meets any of
145      * the following conditions:
146      * <ul>
147      *     <li>when it matches {@code text/*} or {@code application/x-www-form-urlencoded}</li>
148      *     <li>when its charset has been specified</li>
149      *     <li>when its subtype is {@code "xml"} or {@code "json"}</li>
150      *     <li>when its subtype ends with {@code "+xml"} or {@code "+json"}</li>
151      * </ul>
152      * @param length the maximum length of the preview
153      * @param defaultCharset the default charset used when a charset is not specified in the
154      *                       {@code "content-type"} header
155      */
156     public AbstractServiceBindingBuilder contentPreview(int length, Charset defaultCharset) {
157         return contentPreviewerFactory(ContentPreviewerFactory.ofText(length, defaultCharset));
158     }
159 
160     /**
161      * Sets the {@link ContentPreviewerFactory} for an HTTP request/response of a {@link Service}.
162      */
163     public AbstractServiceBindingBuilder contentPreviewerFactory(ContentPreviewerFactory factory) {
164         requestContentPreviewerFactory(factory);
165         responseContentPreviewerFactory(factory);
166         return this;
167     }
168 
169     /**
170      * Sets the format of this {@link Service}'s access log. The specified {@code accessLogFormat} would be
171      * parsed by {@link AccessLogWriter#custom(String)}.
172      */
173     public AbstractServiceBindingBuilder accessLogFormat(String accessLogFormat) {
174         return accessLogWriter(AccessLogWriter.custom(requireNonNull(accessLogFormat, "accessLogFormat")),
175                                true);
176     }
177 
178     /**
179      * Sets the access log writer of this {@link Service}. If not set, {@link ServerConfig#accessLogWriter()}
180      * is used.
181      *
182      * @param shutdownOnStop whether to shut down the {@link AccessLogWriter} when the {@link Server} stops
183      */
184     public AbstractServiceBindingBuilder accessLogWriter(AccessLogWriter accessLogWriter,
185                                                          boolean shutdownOnStop) {
186         this.accessLogWriter = requireNonNull(accessLogWriter, "accessLogWriter");
187         shutdownAccessLogWriterOnStop = shutdownOnStop;
188         return this;
189     }
190 
191     /**
192      * Decorates a {@link Service} with the specified {@code decorator}.
193      *
194      * @param decorator the {@link Function} that decorates the {@link Service}
195      * @param <T> the type of the {@link Service} being decorated
196      * @param <R> the type of the {@link Service} {@code decorator} will produce
197      */
198     public <T extends Service<HttpRequest, HttpResponse>, R extends Service<HttpRequest, HttpResponse>>
199     AbstractServiceBindingBuilder decorator(Function<T, R> decorator) {
200         requireNonNull(decorator, "decorator");
201 
202         @SuppressWarnings("unchecked")
203         final Function<Service<HttpRequest, HttpResponse>, Service<HttpRequest, HttpResponse>> castDecorator =
204                 (Function<Service<HttpRequest, HttpResponse>, Service<HttpRequest, HttpResponse>>) decorator;
205 
206         if (this.decorator != null) {
207             this.decorator = this.decorator.andThen(castDecorator);
208         } else {
209             this.decorator = castDecorator;
210         }
211 
212         return this;
213     }
214 
215     abstract void serviceConfigBuilder(ServiceConfigBuilder serviceConfigBuilder);
216 
217     final void build0(Service<HttpRequest, HttpResponse> service) {
218         final List<Route> routes = buildRouteList();
219 
220         for (Route route : routes) {
221             final ServiceConfigBuilder serviceConfigBuilder =
222                     new ServiceConfigBuilder(route, decorate(service));
223             if (requestTimeoutMillis != null) {
224                 serviceConfigBuilder.requestTimeoutMillis(requestTimeoutMillis);
225             }
226             if (maxRequestLength != null) {
227                 serviceConfigBuilder.maxRequestLength(maxRequestLength);
228             }
229             if (verboseResponses != null) {
230                 serviceConfigBuilder.verboseResponses(verboseResponses);
231             }
232             if (requestContentPreviewerFactory != null) {
233                 serviceConfigBuilder.requestContentPreviewerFactory(requestContentPreviewerFactory);
234             }
235             if (responseContentPreviewerFactory != null) {
236                 serviceConfigBuilder.responseContentPreviewerFactory(responseContentPreviewerFactory);
237             }
238             if (accessLogWriter != null) {
239                 serviceConfigBuilder.accessLogWriter(accessLogWriter, shutdownAccessLogWriterOnStop);
240             }
241             serviceConfigBuilder(serviceConfigBuilder);
242         }
243     }
244 
245     private Service<HttpRequest, HttpResponse> decorate(Service<HttpRequest, HttpResponse> service) {
246         if (decorator == null) {
247             return service;
248         }
249 
250         return service.decorate(decorator);
251     }
